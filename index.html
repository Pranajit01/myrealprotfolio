sensor_pipeline.py 
import tensorflow as tf
import numpy as np

# DrainMind early-alert pipeline logic
model = tf.keras.models.load_model("drain_mind")

def predict_flood_risk(sensor_log):
    features = np.array(sensor_log).reshape(1, -1)
    prob = model.predict(features)[0][0]
    
    if prob >= 0.85:
        return {"alert": "CRITICAL", "val": prob}
    return {"alert": "NORMAL", "val": prob}
